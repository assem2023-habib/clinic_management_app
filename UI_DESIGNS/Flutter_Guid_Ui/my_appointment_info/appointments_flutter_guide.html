enum AppointmentStatus {
  confirmed,   // مؤكد  — أخضر
  pending,     // قيد الانتظار — برتقالي/وردي
  completed,   // مكتمل — secondary
  cancelled;   // ملغي  — error

  String label() => switch(this){
    confirmed  => 'مؤكد',
    pending    => 'قيد الانتظار',
    completed  => 'مكتمل',
    cancelled  => 'ملغي',
  };

  Color color() => switch(this){
    confirmed  => AppColors.primary,
    pending    => AppColors.tertiary,
    completed  => AppColors.secondary,
    cancelled  => AppColors.error,
  };

  IconData icon() => switch(this){
    confirmed  => Icons.check_circle,
    pending    => Icons.schedule,
    completed  => Icons.done_all,
    cancelled  => Icons.cancel,
  };
}

Timeline الرأسي — Custom Painter

الـ Timeline هو الخط الأخضر الرأسي مع النقاط الدائرية. يُرسم بـ CustomPainter ويتغير شكل كل نقطة ولونها حسب الـ status.

🟢

مؤكد (Confirmed)

نقطة كبيرة خضراء مع glow، ring أبيض حول النقطة

⚪

انتظار (Pending)

نقطة أصغر رمادية بدون glow

⬜

مكتمل (Done)

نقطة صغيرة جداً باهتة شفافة

lib/widgets/timeline_painter.dartdart
class TimelinePainter extends CustomPainter {
  final AppointmentStatus status;
  final bool isFirst;
  final bool isLast;

  const TimelinePainter({
    required this.status,
    this.isFirst = false,
    this.isLast  = false,
  });

  @override
  void paint(Canvas canvas, Size size) {
    const lineX = 19.0;   // موقع الخط أفقياً من اليمين
    const dotY  = 22.0;   // موقع النقطة عمودياً

    // ── رسم الخط الرأسي ─────────────────────
    if (!isLast) {
      final linePaint = Paint()
        ..shader = LinearGradient(
          begin: Alignment.topCenter,
          end:   Alignment.bottomCenter,
          colors: [
            const Color(0xFF006D44),
            const Color(0xFF006D44).withOpacity(.15),
          ],
        ).createShader(Rect.fromLTWH(lineX, dotY, 2, size.height - dotY))
        ..strokeWidth = 2
        ..style = PaintingStyle.stroke;

      canvas.drawLine(
        Offset(lineX, dotY + 8),
        Offset(lineX, size.height),
        linePaint,
      );
    }

    // ── رسم النقطة حسب الـ status ───────────
    switch (status) {
      case AppointmentStatus.confirmed:
        // النقطة الكبيرة الخضراء مع glow
        canvas.drawCircle(
          Offset(lineX, dotY),
          7,
          Paint()
            ..color = const Color(0xFF80D8A6)
            ..maskFilter = const MaskFilter.blur(BlurStyle.normal, 4),
        );
        // ring خارجي
        canvas.drawCircle(
          Offset(lineX, dotY), 7,
          Paint()..color = const Color(0xFF80D8A6),
        );
        canvas.drawCircle(
          Offset(lineX, dotY), 10,
          Paint()
            ..color = AppColors.background
            ..style = PaintingStyle.stroke
            ..strokeWidth = 3,
        );

      case AppointmentStatus.pending:
        canvas.drawCircle(
          Offset(lineX, dotY), 5,
          Paint()..color = const Color(0xFF88938A),  // outline
        );

      case AppointmentStatus.completed:
        canvas.drawCircle(
          Offset(lineX, dotY), 4,
          Paint()..color = const Color(0xFFBEC9BF).withOpacity(.6),
        );

      case AppointmentStatus.cancelled:
        canvas.drawCircle(
          Offset(lineX, dotY), 5,
          Paint()..color = const Color(0xFFFFB4AB).withOpacity(.7),
        );
    }
  }

  @override
  bool shouldRepaint(covariant CustomPainter old) => false;
}

بطاقة الموعد — Appointment Card

البطاقة تدعم ثلاثة أنماط مرئية مختلفة: عادية، featured (gradient خاص مع dot pattern)، وpast (شفافية مخفضة). كلها تشترك في نفس الـ widget لكن مع خصائص مختلفة.

✅ بطاقة عادية (Confirmed)

Glass bg بسيط، border أبيض 7%، hover يضيف shadow أخضر

⭐ بطاقة Featured (Pending)

Radial gradient + dot pattern + border أخضر + inset glow

lib/widgets/appointment_card.dartdart
class AppointmentCard extends StatefulWidget {
  final Appointment appointment;
  const AppointmentCard({super.key, required this.appointment});
  @override
  State<AppointmentCard> createState() => _AppointmentCardState();
}

class _AppointmentCardState extends State<AppointmentCard>
    with SingleTickerProviderStateMixin {

  bool _hov = false;
  late AnimationController _entrCtrl;
  late Animation<double> _opacity;
  late Animation<Offset> _slide;

  @override
  void initState() {
    super.initState();
    _entrCtrl = AnimationController(vsync:this, duration:const Duration(ms:550));
    _opacity  = CurvedAnimation(parent:_entrCtrl, curve:Curves.easeOut);
    _slide    = Tween<Offset>(begin:const Offset(0,.25), end:Offset.zero)
        .animate(CurvedAnimation(parent:_entrCtrl, curve:Curves.easeOut));
    Future.delayed(const Duration(ms:120), ()=>mounted?_entrCtrl.forward():null);
  }

  @override
  void dispose(){ _entrCtrl.dispose(); super.dispose(); }

  BoxDecoration _decoration() {
    final apt = widget.appointment;
    if (apt.status == AppointmentStatus.completed) {
      return BoxDecoration(
        color: const Color(0xFF001207).withOpacity(.35),
        borderRadius: BorderRadius.circular(16),
      );
    }
    if (apt.isFeatured) {
      return BoxDecoration(
        borderRadius: BorderRadius.circular(16),
        gradient: RadialGradient(
          colors: [
            const Color(0xFF006D44).withOpacity(.25),
            const Color(0xFF002111).withOpacity(.5),
          ],
        ),
        border: Border.all(color: const Color(0xFF80D8A6).withOpacity(.22)),
        boxShadow: [
          BoxShadow(
            color: const Color(0xFF80D8A6).withOpacity(.12),
            blurRadius: 20, spreadRadius: -4,
          ),
        ],
      );
    }
    // عادية glass
    return BoxDecoration(
      color: const Color(0xFF012B17).withOpacity(.42),
      borderRadius: BorderRadius.circular(16),
      border: Border.all(color: Colors.white.withOpacity(.07)),
      boxShadow: _hov ? [BoxShadow(
        color: const Color(0xFF006D44).withOpacity(.2),
        blurRadius: 36,
      )] : [],
    );
  }

  @override
  Widget build(BuildContext context) {
    return FadeTransition(opacity:_opacity,
      child: SlideTransition(position:_slide,
        child: MouseRegion(
          onEnter:(_)=>setState(()=>_hov=true),
          onExit: (_)=>setState(()=>_hov=false),
          child: AnimatedScale(
            scale: _hov ? 1.015 : 1.0,
            duration:const Duration(ms:250),
            child: ClipRRect(
              borderRadius: BorderRadius.circular(16),
              child: BackdropFilter(
                filter:ImageFilter.blur(sigmaX:16,sigmaY:16),
                child: AnimatedContainer(
                  duration:const Duration(ms:260),
                  decoration: _decoration(),
                  padding:const EdgeInsets.all(16),
                  child: Column(children:[
                    _buildTop(),
                    _buildDivider(),
                    _buildBottom(),
                  ]),
                ),
              ),
            ),
          ),
        ),
      ),
    );
  }
}

الفلترة والبحث — Filter Logic

البحث والفلترة مشتركان في getter واحد. نستخدم Riverpod أو setState بسيطة.

منطق الفلترةdart
// في _AppointmentsScreenState

List<dynamic> get _filteredItems {
  final q = _searchQ.trim().toLowerCase();
  final filtered = kAppointments.where((a) {
    final statusMatch = _filter == 'all'
        || (_filter == 'upcoming'  && (a.status==confirmed||a.status==pending))
        || (_filter == 'done'      && a.status==completed)
        || (_filter == 'cancelled' && a.status==cancelled);

    final textMatch = q.isEmpty
        || a.doctorName.toLowerCase().contains(q)
        || a.specialty.toLowerCase().contains(q);

    return statusMatch && textMatch;
  }).toList();

  // أضف فواصل الشهر تلقائياً حسب groupBy
  return _insertMonthDividers(filtered);
}

List<dynamic> _insertMonthDividers(List<Appointment> list) {
  final result = <dynamic>[];
  String? lastMonth;
  for (final a in list) {
    final m = '${a.dateTime.month}/${a.dateTime.year}';
    if (m != lastMonth) {
      lastMonth = m;
      result.add(MonthLabel(a.dateTime));  // سيُعرض كـ MonthDivider
    }
    result.add(a);
  }
  return result;
}

// ── Rising Particles ──────────────────────────
class RisingParticles extends StatefulWidget {
  final int count;
  const RisingParticles({super.key, this.count = 30});
  @override
  State<RisingParticles> createState() => _RisingParticlesState();
}

class _RisingParticlesState extends State<RisingParticles>
    with SingleTickerProviderStateMixin {

  late AnimationController _ctrl;
  final _rand = Random();
  late List<_RisingParticle> _particles;

  @override
  void initState() {
    super.initState();
    _ctrl = AnimationController(vsync: this, duration: const Duration(seconds: 1))
      ..addListener(_tick)..repeat();
    _particles = List.generate(widget.count, (_) => _makeParticle());
  }

  _RisingParticle _makeParticle() => _RisingParticle(
    x:       _rand.nextDouble(),
    progress: _rand.nextDouble(),
    size:    1 + _rand.nextDouble() * 3,
    speed:   (10 + _rand.nextDouble() * 10) / 1000,
  );

  void _tick() {
    setState(() {
      for (var i = 0; i < _particles.length; i++) {
        _particles[i].progress += _particles[i].speed;
        if (_particles[i].progress >= 1.0) {
          _particles[i] = _makeParticle()..progress = 0;
        }
      }
    });
  }

  @override
  void dispose() { _ctrl.dispose(); super.dispose(); }

  @override
  Widget build(BuildContext context) =>
    RepaintBoundary(  // ← مهم للأداء
      child: CustomPaint(
        painter: _RisingPainter(_particles),
        size: Size.infinite,
      ),
    );
}

class _RisingParticle {
  double x, progress, size, speed;
  _RisingParticle({required this.x, required this.progress,
                    required this.size, required this.speed});
}

class _RisingPainter extends CustomPainter {
  final List<_RisingParticle> particles;
  const _RisingPainter(this.particles);

  @override
  void paint(Canvas canvas, Size size) {
    for (final p in particles) {
      final t = p.progress;
      final y = size.height * (1 - t);  // يرتفع من الأسفل للأعلى
      // fade in 10%، ثابت حتى 90%، ثم fade out
      final op = t < .1 ? t / .1 : (t > .9 ? (1 - t) / .1 : 1.0);
      canvas.drawCircle(
        Offset(size.width * p.x, y), p.size / 2,
        Paint()..color = const Color(0xFF80D8A6).withOpacity(op * .4),
      );
    }
  }

  @override
  bool shouldRepaint(covariant CustomPainter _) => true;
}

❓ أسئلة شائعة

كيف أجعل الـ Timeline يمتد بطول الكارد بالضبط؟

غلّف الـ Row الذي يضم عمود الـ Timeline والكارد بـ IntrinsicHeight. هذا يجبر كلا العمودين على أخذ ارتفاع الأطول بينهما.

لماذا التاريخ "غداً" يظهر باللون الأخضر بينما "الخميس" رمادي؟

نقارن تاريخ الموعد بـ DateTime.now(). إذا كان خلال 48 ساعة القادمة يأخذ لون primary، وإلا onSurfaceVariant. final isUpcoming = appt.dateTime.difference(DateTime.now()).inHours < 48;

كيف أفتح Reschedule Sheet فوق Options Sheet؟

أغلق الـ Options Sheet بـ Navigator.pop(context) أولاً، ثم افتح الـ Reschedule Sheet بـ showModalBottomSheet مع useRootNavigator: true لضمان الظهور فوق كل شيء.

كيف أنسّق التاريخ بالعربية "غداً • 15 أكتوبر"؟

استخدم مكتبة intl: DateFormat('d MMMM yyyy', 'ar').format(dateTime). للتحقق من "غداً": قارن DateUtils.isSameDay(dateTime, DateTime.now().add(Duration(days:1))).

ما الفرق بين BackdropFilter للـ Orbs وللـ Glass Cards؟

الـ Orbs لا تحتاج BackdropFilter فعلياً — نستخدم filter: blur(80px) في CSS أو مجرد Container مع color.withOpacity(0.15) في Flutter لأن الـ orb نفسها شفافة. أما Glass Cards فتحتاج BackdropFilter الحقيقي لطمس ما خلفها.
