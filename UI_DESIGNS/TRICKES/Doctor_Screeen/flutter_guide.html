lib/widgets/particle_background.dartdart
import 'dart:math';
import 'package:flutter/material.dart';

// ── نموذج الجسيم ─────────────────────────────
class Particle {
  Offset position;
  Offset target;
  double size;
  double progress;  // 0..1 للـ lifecycle
  double speed;

  Particle({
    required this.position, required this.target,
    required this.size,     required this.speed,
  }) : progress = 0;
}

// ── CustomPainter للرسم ───────────────────────
class ParticlePainter extends CustomPainter {
  final List<Particle> particles;
  final double pulseProgress;  // 0..1 للنبضة المركزية

  ParticlePainter(this.particles, this.pulseProgress);

  @override
  void paint(Canvas canvas, Size size) {
    // 1. رسم النبضة المركزية
    final center = Offset(size.width / 2, size.height / 2);
    final pulseRadius = pulseProgress * size.longestSide * 1.5;
    final pulseOpacity = (1.0 - pulseProgress) * 0.15;
    final pulsePaint = Paint()
      ..style = PaintingStyle.stroke
      ..strokeWidth = 1.5
      ..color = const Color(0xFF40E78C).withOpacity(pulseOpacity);
    canvas.drawCircle(center, pulseRadius, pulsePaint);

    // 2. رسم كل جسيم
    for (final p in particles) {
      // حساب الموقع الحالي (linear interpolation)
      final t = p.progress;
      final current = Offset(
        p.position.dx + (p.target.dx - p.position.dx) * t,
        p.position.dy + (p.target.dy - p.position.dy) * t,
      );

      // حساب الشفافية (fade in ثم fade out)
      final opacity = t < 0.5
          ? (t / 0.5) * 0.6
          : (1.0 - (t - 0.5) / 0.5) * 0.6;

      final paint = Paint()
        ..color = const Color(0xFF10B981).withOpacity(opacity)
        ..maskFilter = const MaskFilter.blur(BlurStyle.normal, 2); // glow

      canvas.drawCircle(current, p.size / 2, paint);
    }
  }

  @override
  bool shouldRepaint(covariant CustomPainter old) => true;
}

// ── Widget رئيسي ──────────────────────────────
class ParticleBackground extends StatefulWidget {
  const ParticleBackground({super.key});

  @override
  State<ParticleBackground> createState() => _ParticleBackgroundState();
}

class _ParticleBackgroundState extends State<ParticleBackground>
    with TickerProviderStateMixin {

  final _rand = Random();
  final _particles = <Particle>[];
  late AnimationController _controller;
  late AnimationController _pulseCtrl;
  late Animation<double> _pulseAnim;

  @override
  void initState() {
    super.initState();

    // Controller عام يُحدّث الرسم كل frame
    _controller = AnimationController(
      vsync: this,
      duration: const Duration(seconds: 1),
    )..addListener(_updateParticles)..repeat();

    // Controller النبضة المركزية (6 ثواني)
    _pulseCtrl = AnimationController(
      vsync: this,
      duration: const Duration(seconds: 6),
    );
    _pulseAnim = CurvedAnimation(parent: _pulseCtrl, curve: Curves.easeOut);
    _pulseCtrl.repeat();
  }

  void _updateParticles() {
    if (!mounted) return;
    setState(() {
      // أضف جسيمات جديدة حتى الحد الأقصى
      while (_particles.length < 25) {
        _spawnParticle();
      }
      // حدّث progress كل جسيم
      for (var i = _particles.length - 1; i >= 0; i--) {
        _particles[i].progress += _particles[i].speed;
        if (_particles[i].progress >= 1.0) {
          _particles.removeAt(i);  // ميت — احذفه
        }
      }
    });
  }

  void _spawnParticle() {
    final sw = MediaQuery.of(context).size.width;
    final sh = MediaQuery.of(context).size.height;
    final sx = _rand.nextDouble() * sw;
    final sy = _rand.nextDouble() * sh;
    _particles.add(Particle(
      position: Offset(sx, sy),
      target: Offset(
        sx + (_rand.nextDouble() - 0.5) * 200,
        sy + (_rand.nextDouble() - 0.5) * 200,
      ),
      size: _rand.nextDouble() * 4 + 2,
      speed: 0.003 + _rand.nextDouble() * 0.004,
    ));
  }

  @override
  void dispose() {
    _controller.dispose();
    _pulseCtrl.dispose();
    super.dispose();
  }

  @override
  Widget build(BuildContext context) {
    return Positioned.fill(
      child: AnimatedBuilder(
        animation: _pulseAnim,
        builder: (_, __) => CustomPaint(
          painter: ParticlePainter(_particles, _pulseAnim.value),
        ),
      ),
    );
  }
}

بطاقة الطبيب — Glassmorphism Card

القلب النابض للشاشة. البطاقة تحتوي على: تأثير زجاجي (BackdropFilter)، صورة الطبيب مع نقطة الإتاحة المتوهجة، تقييم النجوم، المعلومات، والزر. عند hover على الويب أو الضغط على الموبايل يكون هناك تأثير scale وglow.

📱 على الموبايل (Mobile)

• GestureDetector مع onTapDown / onTapUp
• AnimatedScale بين 1.0 و 0.98
• لا توجد hover حقيقية على الموبايل

🖥️ على الويب (Web/Desktop)

• MouseRegion لاكتشاف الـ hover
• AnimatedScale إلى 1.02
• AnimatedContainer يضيف box shadow خضراء

lib/widgets/doctor_card.dartdart
class DoctorCard extends StatefulWidget {
  final Doctor doctor;
  final Duration animDelay;

  const DoctorCard({
    super.key, required this.doctor, required this.animDelay,
  });

  @override
  State<DoctorCard> createState() => _DoctorCardState();
}

class _DoctorCardState extends State<DoctorCard>
    with SingleTickerProviderStateMixin {

  bool _isHovered = false;
  bool _isPressed = false;

  // Entrance animation (slideUp + fade)
  late AnimationController _entranceCtrl;
  late Animation<double> _opacity;
  late Animation<Offset> _slide;

  @override
  void initState() {
    super.initState();
    _entranceCtrl = AnimationController(
      vsync: this,
      duration: const Duration(milliseconds: 600),
    );
    _opacity = CurvedAnimation(parent: _entranceCtrl, curve: Curves.easeOut);
    _slide = Tween<Offset>(
      begin: const Offset(0, 0.3),   // من أسفل
      end: Offset.zero,
    ).animate(CurvedAnimation(parent: _entranceCtrl, curve: Curves.easeOut));

    // تأخير الظهور حسب ترتيب البطاقة
    Future.delayed(widget.animDelay, () {
      if (mounted) _entranceCtrl.forward();
    });
  }

  @override
  void dispose() {
    _entranceCtrl.dispose();
    super.dispose();
  }

  @override
  Widget build(BuildContext context) {
    return FadeTransition(           // ← fade in عند الدخول
      opacity: _opacity,
      child: SlideTransition(       // ← slide up عند الدخول
        position: _slide,
        child: MouseRegion(         // ← hover للويب والـ desktop
          onEnter: (_) => setState(() => _isHovered = true),
          onExit:  (_) => setState(() => _isHovered = false),
          child: GestureDetector(   // ← اكتشاف اللمس
            onTapDown: (_) => setState(() => _isPressed = true),
            onTapUp:   (_) => setState(() => _isPressed = false),
            onTapCancel: ()  => setState(() => _isPressed = false),
            child: AnimatedScale(   // ← تأثير التكبير/التصغير
              scale: _isPressed ? 0.98 : (_isHovered ? 1.02 : 1.0),

            : [],
      ),
      child: ClipRRect(
        borderRadius: BorderRadius.circular(20),
        child: BackdropFilter(        // ← تأثير الزجاج
          filter: ImageFilter.blur(sigmaX: 12, sigmaY: 12),
          child: Container(
            padding: const EdgeInsets.all(20),
            decoration: BoxDecoration(
              color: AppColors.glassBg,      // rgba(1,43,23, 0.4)
              borderRadius: BorderRadius.circular(20),
              border: Border.all(
                color: AppColors.glassBorder,  // أبيض 8%
              ),
            ),
            child: Column(
              crossAxisAlignment: CrossAxisAlignment.stretch,
              children: [
                _buildCardHeader(),
                const SizedBox(height: 12),
                _buildLocationRow(),
                const SizedBox(height: 6),
                _buildBio(),
                const SizedBox(height: 14),
                _buildBookButton(),
              ],
            ),
          ),
        ),
      ),

  }

  // ─── Header: صورة + اسم + تخصص ──────────────
  Widget _buildCardHeader() {

class _PulsingDotState extends State<_PulsingDot>
    with SingleTickerProviderStateMixin {

  late AnimationController _ctrl;
  late Animation<double> _opacityAnim;

  @override
  void initState() {
    super.initState();
    _ctrl = AnimationController(
      vsync: this,
      duration: const Duration(seconds: 2),
    );
    _opacityAnim = TweenSequence([
      TweenSequenceItem(
        tween: Tween(begin: 1.0, end: 0.3)

class _GlowButtonState extends State<GlowButton> {
  bool _hov = false, _pressed = false;

class TiltCard extends StatefulWidget {
  final Widget child;
  const TiltCard({super.key, required this.child});
  @override
  State<TiltCard> createState() => _TiltCardState();
}

class _TiltCardState extends State<TiltCard> {
  double _rotX = 0, _rotY = 0;
  bool   _hovered = false;

  void _onHover(PointerHoverEvent event, BoxConstraints box) {
    if (!_hovered) return;
    final local = event.localPosition;
    final dx = (local.dx / box.maxWidth  - 0.5) * 2;  // -1..1
    final dy = (local.dy / box.maxHeight - 0.5) * 2;  // -1..1
    setState(() {
      _rotY =  dx * 0.06;   // max ~3.4° rotation
      _rotX = -dy * 0.06;
    });
  }

  @override
  Widget build(BuildContext context) {
    return LayoutBuilder(
      builder: (context, constraints) {
        return MouseRegion(
          onEnter: (_) => setState(() => _hovered = true),
          onExit:  (_) => setState(() {
            _hovered = false;
            _rotX = 0; _rotY = 0;  // reset عند الخروج
          }),
          onHover: (e) => _onHover(e, constraints),
          child: AnimatedContainer(
            duration: const Duration(milliseconds: 300),
            transform: (Matrix4.identity()
              ..setEntry(3, 2, 0.001)  // perspective
              ..rotateX(_rotX)
              ..rotateY(_rotY)),
            transformAlignment: Alignment.center,
            child: widget.child,
          ),
        );
      },
    );
  }
}

❓ أسئلة شائعة

BackdropFilter لا يعمل على الكارد، لماذا؟

تأكد من أن ClipRRect أو ClipRect يُغلّف الـ BackdropFilter. بدونه، يُطبّق الـ blur على كل الشاشة. أيضاً تأكد أن sigmaX وsigmaY ليسا 0.

كيف أجعل الـ AppBar شفافاً ويرى الـ particles خلفه؟

في الـ Scaffold اضبط extendBodyBehindAppBar: true وجعل backgroundColor للـ AppBar شفافاً. الـ body سيمتد تحت الـ AppBar وستظهر الجسيمات خلفه.

لماذا تأثير الـ 3D Tilt يُبطئ التطبيق؟

تأكد من تغليف الكارد بـ RepaintBoundary. الـ Transform مع Matrix4 يُنشئ layer جديدة، والـ RepaintBoundary يمنع إعادة رسم الـ widgets المجاورة.

هل يمكن استخدام Rive بدلاً من CustomPainter للجسيمات؟

نعم، Rive خيار ممتاز للأنيميشن المعقد. لكن للجسيمات البسيطة، CustomPainter أخف وزناً وأسهل تخصيصاً. Rive يُفيد أكثر للـ character animations والـ illustrations.

كيف أضيف دعم الـ Light Mode؟

أنشئ AppColorsLight بنفس البنية مع الألوان الفاتحة من Design System (الخلفية #f0fdf4، السطح #ffffff). استخدم ThemeMode.system في MaterialApp ليتابع إعداد الجهاز تلقائياً.

Blur الخلفية عند فتح الـ Bottom Sheet

في الشاشة الجديدة، عند فتح الـ Bottom Sheet تظهر خلفية ضبابية (blur + تعتيم). لتحقيق هذا في Flutter نستخدم Overlay مخصصاً مع BackdropFilter بدلاً من الـ barrier الافتراضي.

Bottom Sheet بـ Blur Backdrop مخصصdart
// ── نسخة متقدمة من _openDoctorBottomSheet ────
void _openDoctorBottomSheetWithBlur(
  BuildContext context, {
  required Doctor doctor,
}) {
  late OverlayEntry blurOverlay;

  // 1. نُنشئ طبقة الـ blur أولاً
  blurOverlay = OverlayEntry(
    builder: (_) => GestureDetector(
      onTap: () {
        Navigator.pop(context);   // يُغلق الـ Sheet
        blurOverlay.remove();
      },
      child: BackdropFilter(
        filter: ImageFilter.blur(sigmaX: 4, sigmaY: 4),
        child: Container(
          color: AppColors.background.withOpacity(0.7),
        ),
      ),
    ),
  );

  // 2. نُضيفها للـ Overlay
  Overlay.of(context).insert(blurOverlay);

  // 3. نفتح الـ Sheet فوقها
  showModalBottomSheet(
    context: context,
    backgroundColor: Colors.transparent,
    barrierColor: Colors.transparent,  // ← شفاف (لأن الـ blur overlay تتولى ذلك)
    isScrollControlled: true,
    enableDrag: true,
    builder: (ctx) => DoctorOptionsSheet(doctor: doctor),
  ).whenComplete(() => blurOverlay.remove());  // ← حذف الـ blur عند الإغلاق
}
